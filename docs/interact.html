4.1 DT: Datatables

library(readr)
library(dplyr)
library(sf)
fl_accidents <- read_csv("data/fl_accidents.csv") 
library(DT)
datatable(fl_accidents)
datatable(fl_accidents, 
          class = "compact", 
          caption = "Fatal motor vehicle accidents in Florida the week of Hurricane Irma",
          colnames = c("County FIPS", "Date",
                       "Latitude", "Longitude", "# fatalities"))
